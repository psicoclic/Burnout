    // --- COMPONENTE DE RESPIRACIÓN SOS (4-7-8) ---
    function BreathingApp({ severity }) {
      const [isActive, setIsActive] = useState(false);
      const [seconds, setSeconds] = useState(0);

      useEffect(() => {
        let interval = null;
        if (isActive) {
          interval = setInterval(() => {
            setSeconds(s => (s + 1) % 19); // Ciclo de 19 segundos
          }, 1000);
        } else {
          setSeconds(0);
          clearInterval(interval);
        }
        return () => clearInterval(interval);
      }, [isActive]);

      let phase = "Listo";
      let instruction = "Pulsa el botón de abajo para comenzar a modular tu pulso cardíaco";
      let scale = 1;
      let colorClass = "bg-slate-800 text-white";
      let ringColor = "border-slate-200";

      if (isActive) {
        if (seconds < 4) {
          phase = "Inhala";
          instruction = "Toma aire de forma suave y profunda por la nariz.";
          scale = 1 + (seconds / 4) * 0.45; 
          colorClass = "bg-sky-500 text-white";
          ringColor = "border-sky-300";
        } else if (seconds < 11) {
          phase = "Sostiene";
          instruction = "Mantén el aire inhalado e inunda tus células.";
          scale = 1.45;
          colorClass = "bg-indigo-600 text-white";
          ringColor = "border-indigo-400";
        } else {
          phase = "Exhala";
          instruction = "Saca el aire de forma prolongada por la boca.";
          scale = 1.45 - ((seconds - 11) / 8) * 0.45; 
          colorClass = "bg-emerald-500 text-white";
          ringColor = "border-emerald-300";
        }
      }

            frequency: "Frecuencia recomendada: 1 vez al día (como higiene nocturna)",

            schedule: "Rutina propuesta: Aplícalo por la noche para vaciar tu tensión mental acumulada del día."
          };
        }
      }, [severity]);

              ></div>
              <div 
                className={cn("absolute w-32 h-32 rounded-full border opacity-50 breathe-circle", ringColor)}
                style={{ transform: `scale(${scale * 1.15})` }}

                    {seconds < 4 ? `${seconds + 1}s / 4s` : seconds < 11 ? `${seconds - 3}s / 7s` : `${seconds - 10}s / 8s`}
